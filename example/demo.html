components/StatCard.tsx
TSX
// These are YOUR components — styled however you want.
// The package just calls them with the props the LLM provides.

const StatCard = ({ title, value, subtitle }: {
  title: string;
  value: string;
  subtitle?: string;
}) => (
  <div className="stat-card">
    <p className="label">{title}</p>
    <h2 className="value">{value}</h2>
    {subtitle && <small>{subtitle}</small>}
  </div>
);

const ProConTable = ({ title, pros, cons }) => (
  <div className="procon">
    <h4>{title}</h4>
    <div className="grid">
      <ul>{pros.map(p => <li>✅ {p}</li>)}</ul>
      <ul>{cons.map(c => <li>❌ {c}</li>)}</ul>
    </div>
  </div>
);
registry.ts
TS
import { ComponentRegistry } from 'react-generative-ui';
import { StatCard } from './StatCard';
import { ProConTable } from './ProConTable';
import { AlertBox } from './AlertBox';
import { ModelLeaderboard } from './ModelLeaderboard'; // ← your existing ones!
import { CodeSnippet } from './CodeSnippet';

// The registry is just a plain object: name → component
export const registry: ComponentRegistry = {
  StatCard,
  ProConTable,
  AlertBox,
  ModelLeaderboard,
  CodeSnippet,
};

The key insight: the string key must match what you tell the LLM to use. The LLM outputs "componentName": "StatCard" and the registry maps that string to your actual StatCard component.

api/chat.ts
TS
import { buildSystemPrompt } from 'react-generative-ui';
import { registry } from './registry';

// buildSystemPrompt() auto-generates the LLM instructions
const systemPrompt = buildSystemPrompt(registry, {
  StatCard:  { title: "Label", value: "42", subtitle: "optional note" },
  ProConTable: { title: "optional", pros: ["item"], cons: ["item"] },
  AlertBox:  { type: "info|warning|success|error", message: "text" },
});

// The generated prompt instructs the LLM to output:
// {"componentName": "StatCard", "props": {"title":"...", "value":"..."}}

// Use with OpenAI:
const response = await openai.chat.completions.create({
  messages: [
    { role: 'system', content: systemPrompt },
    { role: 'user',   content: userMessage },
  ],
  model: 'gpt-4o',
});
ChatMessage.tsx
TSX
import { GenerativeRenderer, parseBlocks } from 'react-generative-ui';
import { registry } from './registry';

export function ChatMessage({ llmOutput }: { llmOutput: string }) {
  // 1. Parse the raw LLM string into UIBlocks
  const blocks = parseBlocks(llmOutput);

  // 2. Render — that's literally all you do
  return (
    <GenerativeRenderer
      blocks={blocks}
      registry={registry}
    />
  );
}

// Before this package: you'd write 200+ lines of custom parsing logic
// After:  4 lines. Done.
